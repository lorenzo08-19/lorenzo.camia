import random import time def is_prime(num): if num < 2: return False for i in range(2, int(num ** 0.5) + 1): if num % i == 0: return False return True def generare_numero(): while True: numero_casuale = random.randint(1, 1000) print(f"\nNumero casuale: {numero_casuale}") if numero_casuale % 2 == 0: print("Il numero è pari.") else: print("Il numero è dispari.") if is_prime(numero_casuale): print("Il numero è primo.") else: print("Il numero non è primo.") doppio = numero_casuale * 2 print(f"Il doppio del numero è: {doppio}") meta = numero_casuale // 2 print(f"La metà intera del numero è: {meta}") complemento = 1000 - numero_casuale print(f"Il complemento del numero è: {complemento}") precedente = numero_casuale - 1 successivo = numero_casuale + 1 somma = numero_casuale + precedente + successivo print(f"La somma del numero con il precedente e il successivo è: {somma}") moltiplicazione = numero_casuale * 3.14 print(f"Il numero moltiplicato per 3.14 è: {moltiplicazione}") time.sleep(5) generare_numero()
